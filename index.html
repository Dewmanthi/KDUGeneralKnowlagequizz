        const QUESTIONS_PER_LEVEL = 20;

        let shuffledQuizData = []; // All questions, shuffled once at start

        let currentQuestionIndex = 0;

        let totalScore = 0;

        // Get DOM elements

        const startScreen = document.getElementById('start-screen');

        const levelDisplay = document.getElementById('level-display');

        const totalScoreDisplay = document.getElementById('total-score-display');

        const optionsContainer = document.getElementById('options-container');

        const levelStatusHeading = document.getElementById('level-status-heading');

        const closeButton = document.querySelector('.close-button');


        // Function to show custom modal

        function showModal(message) {

            modalMessage.textContent = message;

            customModal.style.display = 'flex';

        }


        // Function to hide custom modal

        function hideModal() {

            customModal.style.display = 'none';

        }


        // Event listeners for modal

        modalOkButton.addEventListener('click', hideModal);

        closeButton.addEventListener('click', hideModal);

        // Function to shuffle array (for options)

        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]]; // Swap elements

            }

            return array;

        }


        // Function to start the quiz

        function startQuiz() {

            // Shuffle all questions once at the beginning

            shuffledQuizData = shuffleArray([...allQuizData]);

            currentQuestionIndex = 0;

            totalScore = 0;

            currentLevel = 1;

            levelScore = 0;

            quizStarted = true;


            // Hide all screens and show quiz section

            startScreen.style.display = 'none';

            levelStatusScreen.style.display = 'none';

            endScreen.style.display = 'none';

            quizSection.style.display = 'block';


            updateScoreDisplays();

            displayQuestion();

        }


        // Function to update score displays

        function updateScoreDisplays() {

            totalScoreDisplay.textContent = `මුළු ලකුණු: ${totalScore}`;

            currentLevelScoreDisplay.textContent = `වත්මන් මට්ටමේ ලකුණු: ${levelScore}`;

            levelDisplay.textContent = `මට්ටම ${currentLevel} / ${Math.ceil(allQuizData.length / QUESTIONS_PER_LEVEL)}`;

        }


        // Function to display the current question and its options

        function displayQuestion() {

            // Check if all questions are done

            if (currentQuestionIndex >= allQuizData.length) {

                endQuiz();

                return;

            }


            const currentQuestionNumberInLevel = (currentQuestionIndex % QUESTIONS_PER_LEVEL) + 1;

            const totalQuestionsInCurrentLevel = Math.min(QUESTIONS_PER_LEVEL, allQuizData.length - ((currentLevel - 1) * QUESTIONS_PER_LEVEL));


            questionCounter.textContent = `ප්‍රශ්නය ${currentQuestionNumberInLevel} / ${totalQuestionsInCurrentLevel}`;


            const question = shuffledQuizData[currentQuestionIndex];

            questionText.textContent = question.question;


            optionsContainer.innerHTML = ''; // Clear previous options

            const shuffledOptions = shuffleArray([...question.options]); // Shuffle options for each question


            shuffledOptions.forEach((option) => {

                const optionLabel = document.createElement('label');

                optionLabel.className = 'option-label';

                optionLabel.innerHTML = `

                    <input type="radio" name="answer" value="${option}" class="form-radio">

                    <span>${option}</span>

                `;

                optionsContainer.appendChild(optionLabel);


                // Add event listener to highlight selected option

                optionLabel.querySelector('input[type="radio"]').addEventListener('change', () => {

                    document.querySelectorAll('.option-label').forEach(label => {

                        label.classList.remove('selected');

                    });

                    if (optionLabel.querySelector('input[type="radio"]').checked) {

                        optionLabel.classList.add('selected');

                    }

                });

            });


            feedback.textContent = '';

            feedback.classList.add('hidden');

            checkAnswerBtn.disabled = false;

            nextQuestionBtn.disabled = true;


            // Enable all radio buttons

            optionsContainer.querySelectorAll('input[type="radio"]').forEach(radio => {

                radio.disabled = false;

            });

        }


        // Function to check the answer

        function checkAnswer() {

            const selectedOption = document.querySelector('input[name="answer"]:checked');

            const feedbackElement = document.getElementById('feedback');


            feedbackElement.classList.remove('hidden', 'feedback-correct', 'feedback-incorrect');

            document.querySelectorAll('.option-label').forEach(label => {

                label.classList.remove('selected', 'correct', 'incorrect');

            });


            if (!selectedOption) {

                showModal('කරුණාකර පිළිතුරක් තෝරන්න.');

                return;

            }


            const userAnswer = selectedOption.value.trim().toLowerCase();

            const correctAnswer = shuffledQuizData[currentQuestionIndex].answer.trim().toLowerCase();


            // Disable all radio buttons after checking

            optionsContainer.querySelectorAll('input[type="radio"]').forEach(radio => {

                radio.disabled = true;

            });


            // Highlight correct and incorrect answers

            optionsContainer.querySelectorAll('.option-label').forEach(label => {

                const radio = label.querySelector('input[type="radio"]');

                if (radio.value.trim().toLowerCase() === correctAnswer) {

                    label.classList.add('correct');

                } else if (radio.checked) {

                    label.classList.add('incorrect');

                }

            });


            if (userAnswer === correctAnswer) {

                totalScore++;

                levelScore++;

                feedbackElement.textContent = 'නිවැරදියි!';

                feedbackElement.classList.add('feedback-correct');

            } else {

                feedbackElement.textContent = `වැරදියි. නිවැරදි පිළිතුර: ${shuffledQuizData[currentQuestionIndex].answer}`;

                feedbackElement.classList.add('feedback-incorrect');

            }


            updateScoreDisplays();

            checkAnswerBtn.disabled = true;

            nextQuestionBtn.disabled = false;

        }


        // Function to move to the next question or check level completion

        function nextQuestion() {

            currentQuestionIndex++;

            // Check if it's the end of the current level

            if (currentQuestionIndex % QUESTIONS_PER_LEVEL === 0 && currentQuestionIndex < allQuizData.length) {

                checkLevelCompletion();

            } else if (currentQuestionIndex >= allQuizData.length) {

                // If all questions are done (last level completed)

                endQuiz();

            } else {

                displayQuestion();

            }

        }


        // Function to check if the user passed the current level

        function checkLevelCompletion() {

            quizSection.style.display = 'none';

            levelStatusScreen.style.display = 'block';


            levelStatusScore.textContent = `මෙම මට්ටමේ ලකුණු: ${levelScore} / ${QUESTIONS_PER_LEVEL}`;

            levelStatusTotalScore.textContent = `මුළු ලකුණු: ${totalScore}`;


            if (levelScore >= PASS_SCORE_THRESHOLD) {

                levelStatusHeading.textContent = `මට්ටම ${currentLevel} සම්පූර්ණයි! 🎉`;

                levelStatusMessage.textContent = `සුභ පැතුම්! ඔබ ඊළඟ මට්ටමට සුදුසුකම් ලබා ගත්තා.`;

                levelStatusPrimaryBtn.textContent = 'ඊළඟ මට්ටම';

                levelStatusPrimaryBtn.onclick = goToNextLevel;

                levelStatusSecondaryBtn.style.display = 'none'; // Hide restart level button

            } else {

                levelStatusHeading.textContent = `මට්ටම ${currentLevel} අසමත්! 😔`;

                levelStatusMessage.textContent = `ඔබට ඊළඟ මට්ටමට යාමට ප්‍රශ්න ${PASS_SCORE_THRESHOLD} ක් නිවැරදිව පිළිතුරු දීමට අවශ්‍ය විය.`;

                levelStatusPrimaryBtn.textContent = 'නැවත උත්සාහ කරන්න';

                levelStatusPrimaryBtn.onclick = restartCurrentLevel;

                levelStatusSecondaryBtn.textContent = 'මුල සිට ආරම්භ කරන්න';

                levelStatusSecondaryBtn.onclick = startQuiz;

                levelStatusSecondaryBtn.style.display = 'inline-block'; // Show restart quiz button

            }

        }


        // Function to go to the next level

        function goToNextLevel() {

            currentLevel++;

            levelScore = 0; // Reset level score for the new level

            levelStatusScreen.style.display = 'none';

            quizSection.style.display = 'block';

            updateScoreDisplays();

            displayQuestion();

        }


        // Function to restart the current level

        function restartCurrentLevel() {

            // Reset currentQuestionIndex to the start of the current level

            currentQuestionIndex = (currentLevel - 1) * QUESTIONS_PER_LEVEL;

            // Deduct current level's score from total score before resetting levelScore

            totalScore -= levelScore;

            levelScore = 0; // Reset level score

            levelStatusScreen.style.display = 'none';

            quizSection.style.display = 'block';

            updateScoreDisplays();

            displayQuestion();

        }


        // Function to end the quiz (when all levels are completed)

        function endQuiz() {

            quizStarted = false;

            quizSection.style.display = 'none';

            levelStatusScreen.style.display = 'none';

            endScreen.style.display = 'block';

            finalScore.textContent = `ඔබ ලකුණු ${totalScore} ක් ලබා ගත්තා. ${allQuizData.length} න්!`;

        }


        // Event Listeners

        startQuizBtn.addEventListener('click', startQuiz);

        checkAnswerBtn.addEventListener('click', checkAnswer);

        nextQuestionBtn.addEventListener('click', nextQuestion);

        restartQuizBtn.addEventListener('click', startQuiz);


        // Initial setup on window load

        window.onload = function() {

            // Ensure start screen is visible and other sections are hidden

            startScreen.style.display = 'block';

            quizSection.style.display = 'none';

            levelStatusScreen.style.display = 'none';

            endScreen.style.display = 'none';

        };
